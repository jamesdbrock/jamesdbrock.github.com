64 /*
65 \brief Internal ascii-to-integer conversion.
66 
67 Parses ascii and returns a (possibly negative) integer.
68 
69 \tparam Int_type The type of integer to be returned.
70 \param begin Pointer to the beginning of the ascii string.
71 \param end Pointer to past-the-end of the ascii string.
72 \return The ascii string represented as an integer of type Int_type.
73 */
74 template<typename Int_type> Int_type atoi(char const* begin, char const* end)
75 {
76  Int_type val(0);
77  bool isnegative(false);
78 
79  if (begin < end && *begin == '-') {
80  isnegative = true;
81  ++begin;
82  }
83 
84  for(; begin<end; ++begin) {
85  val *= 10;
86  val += (Int_type)(*begin - '0');
87  }
88 
89  return isnegative ? -val : val;
90 }

116 /*
117 \brief Internal integer-to-ascii conversion.
118 
119 Writes an integer out as ascii.
120 
121 \tparam Int_type Type of integer to be converted.
122 \param number Value of the integer to be converted.
123 \param buffer Pointer to location for the ascii to be written.
124 \return Pointer to past-the-end of the ascii that was written.
125 */
126 template<typename Int_type> char* itoa(Int_type number, char* buffer)
127 {
128  bool isnegative(false);
129  if (number < 0) {
130  isnegative = true;
131  number = -number;
132  }
133 
134  char*b = buffer;
135  do {
136  *b++ = '0' + (number % 10);
137  number /= 10;
138  } while(number);
139 
140  if (isnegative) {
141  *b++ = '-';
142  }
143 
144  char* e = b - 1;
145 
146  while(e > buffer) { // Reverse the digits in-place.
147  char tmp = *e;
148  *e = *buffer;
149  *buffer = tmp;
150  ++buffer;
151  --e;
152  }
153 
154  return b;
155 }

189 /*
190 \brief Internal ascii-to-decimal conversion.
191 
192 Converts an ascii string to a decimal float, of the form \htmlonly mantissa&times;10<sup>exponent</sup>\endhtmlonly.
193 
194 Non-normalized. The exponent will always be less than or equal to zero.
195 If the decimal float is an integer, the exponent will be zero.
196 
197 \tparam Int_type Signed integer type for the mantissa and exponent.
198 \param begin Pointer to the beginning of the ascii string.
199 \param end Pointer to past-the-end of the ascii string.
200 \param mantissa Reference to storage for the mantissa of the decimal float to be returned.
201 \param exponent Reference to storage for the exponent of the decimal float to be returned.
202 */
203 template<typename Int_type> void atod(char const* begin, char const* end, Int_type& mantissa, Int_type& exponent)
204 {
205  Int_type mantissa_ = 0;
206  Int_type exponent_ = 0;
207  bool isdecimal(false);
208  bool isnegative(false);
209 
210  if (begin < end && *begin == '-') {
211  isnegative = true;
212  ++begin;
213  }
214 
215  for(; begin < end; ++begin) {
216  if (*begin == '.') {
217  isdecimal = true;
218  } else {
219  mantissa_ *= 10;
220  mantissa_ += (*begin - '0');
221  if (isdecimal) --exponent_;
222  }
223  }
224 
225  if (isnegative) mantissa_ = -mantissa_;
226  mantissa = mantissa_;
227  exponent = exponent_;
228 }

230 /*
231 \brief Internal decimal-to-ascii conversion.
232 
233 Converts a decimal float of the form \htmlonly mantissa&times;10<sup>exponent</sup>\endhtmlonly to ascii.
234 
235 Non-normalized. The exponent parameter must be less than or equal to zero.
236 
237 \tparam Int_type Integer type for the mantissa and exponent.
238 \param mantissa The mantissa of the decimal float.
239 \param exponent The exponent of the decimal float. Must be less than or equal to zero.
240 \param buffer Pointer to location for the ascii to be written.
241 \return Pointer to past-the-end of the ascii that was written.
242 */
243 template<typename Int_type> char* dtoa(Int_type mantissa, Int_type exponent, char* buffer)
244 {
245  bool isnegative(false);
246  if (mantissa < 0) {
247  isnegative = true;
248  mantissa = -mantissa;
249  }
250 
251  char*b = buffer;
252  do {
253  *b++ = '0' + (mantissa % 10);
254  mantissa /= 10;
255  if (++exponent == 0) {
256  *b++ = '.';
257  }
258  } while(mantissa > 0 || exponent < 1);
259 
260  if (isnegative) {
261  *b++ = '-';
262  }
263 
264  char* e = b - 1;
265 
266  while(e > buffer) { // Do the reversal in-place instead of making temp buffer. Better cache locality, and we don't have to guess how big to make the temp buffer.
267  char tmp = *e;
268  *e = *buffer;
269  *buffer = tmp;
270  ++buffer;
271  --e;
272  }
273 
274  return b;
275 }

462  void push_back_trailer(bool calculate_checksum = true) {
463  // Calculate and write out the BodyLength.
464  // BodyLength does not include the SOH character after the BodyLength field.
465  // BodyLength does not include the SOH character before the CheckSum field.
466  size_t bodylength = next_ - (body_length_ + 7);
467  for(char* b = body_length_ + 5; b >= body_length_; --b) {
468  *b = '0' + (bodylength % 10);
469  bodylength /= 10;
470  }
471 
472  // write out the CheckSum after optionally calculating it
473  if (calculate_checksum) {
474  size_t checksum(0);
475  char* b = buffer_;
476  while(b < next_) checksum += *b++;
477  checksum = checksum % 256;
478 
479  memcpy(next_, "10=", 3);
480  next_ += 3;
481 
482  for (char* b = next_ + 2; b >= next_; --b) {
483  *b = '0' + (checksum % 10);
484  checksum /= 10;
485  }
486  next_ += 3;
487  *next_++ = '\x01';
488  } else {
489  memcpy(next_, "10=000\x01", 7);
