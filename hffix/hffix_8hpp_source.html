63 /*
64 \brief Internal ascii-to-integer conversion.
65 
66 Parses ascii and returns a (possibly negative) integer.
67 
68 \tparam Int_type The type of integer to be returned.
69 \param begin Pointer to the beginning of the ascii string.
70 \param end Pointer to past-the-end of the ascii string.
71 \return The ascii string represented as an integer of type Int_type.
72 */
73 template<typename Int_type> Int_type atoi(char const* begin, char const* end)
74 {
75  Int_type val(0);
76  bool isnegative(false);
77 
78  if (begin < end && *begin == '-') {
79  isnegative = true;
80  ++begin;
81  }
82 
83  for(; begin<end; ++begin) {
84  val *= 10;
85  val += (Int_type)(*begin - '0');
86  }
87 
88  return isnegative ? -val : val;
89 }

115 /*
116 \brief Internal integer-to-ascii conversion.
117 
118 Writes an integer out as ascii.
119 
120 \tparam Int_type Type of integer to be converted.
121 \param number Value of the integer to be converted.
122 \param buffer Pointer to location for the ascii to be written.
123 \return Pointer to past-the-end of the ascii that was written.
124 */
125 template<typename Int_type> char* itoa(Int_type number, char* buffer)
126 {
127  bool isnegative(false);
128  if (number < 0) {
129  isnegative = true;
130  number = -number;
131  }
132 
133  char*b = buffer;
134  do {
135  *b++ = '0' + (number % 10);
136  number /= 10;
137  } while(number);
138 
139  if (isnegative) {
140  *b++ = '-';
141  }
142 
143  char* e = b - 1;
144 
145  while(e > buffer) { // Reverse the digits in-place.
146  char tmp = *e;
147  *e = *buffer;
148  *buffer = tmp;
149  ++buffer;
150  --e;
151  }
152 
153  return b;
154 }

188 /*
189 \brief Internal ascii-to-decimal conversion.
190 
191 Converts an ascii string to a decimal float, of the form \htmlonly mantissa&times;10<sup>exponent</sup>\endhtmlonly.
192 
193 Non-normalized. The exponent will always be less than or equal to zero.
194 If the decimal float is an integer, the exponent will be zero.
195 
196 \tparam Int_type Signed integer type for the mantissa and exponent.
197 \param begin Pointer to the beginning of the ascii string.
198 \param end Pointer to past-the-end of the ascii string.
199 \param mantissa Reference to storage for the mantissa of the decimal float to be returned.
200 \param exponent Reference to storage for the exponent of the decimal float to be returned.
201 */
202 template<typename Int_type> void atod(char const* begin, char const* end, Int_type& mantissa, Int_type& exponent)
203 {
204  Int_type mantissa_ = 0;
205  Int_type exponent_ = 0;
206  bool isdecimal(false);
207  bool isnegative(false);
208 
209  if (begin < end && *begin == '-') {
210  isnegative = true;
211  ++begin;
212  }
213 
214  for(; begin < end; ++begin) {
215  if (*begin == '.') {
216  isdecimal = true;
217  } else {
218  mantissa_ *= 10;
219  mantissa_ += (*begin - '0');
220  if (isdecimal) --exponent_;
221  }
222  }
223 
224  if (isnegative) mantissa_ = -mantissa_;
225  mantissa = mantissa_;
226  exponent = exponent_;
227 }

229 /*
230 \brief Internal decimal-to-ascii conversion.
231 
232 Converts a decimal float of the form \htmlonly mantissa&times;10<sup>exponent</sup>\endhtmlonly to ascii.
233 
234 Non-normalized. The exponent parameter must be less than or equal to zero.
235 
236 \tparam Int_type Integer type for the mantissa and exponent.
237 \param mantissa The mantissa of the decimal float.
238 \param exponent The exponent of the decimal float. Must be less than or equal to zero.
239 \param buffer Pointer to location for the ascii to be written.
240 \return Pointer to past-the-end of the ascii that was written.
241 */
242 template<typename Int_type> char* dtoa(Int_type mantissa, Int_type exponent, char* buffer)
243 {
244  bool isnegative(false);
245  if (mantissa < 0) {
246  isnegative = true;
247  mantissa = -mantissa;
248  }
249 
250  char*b = buffer;
251  do {
252  *b++ = '0' + (mantissa % 10);
253  mantissa /= 10;
254  if (++exponent == 0) {
255  *b++ = '.';
256  }
257  } while(mantissa > 0 || exponent < 1);
258 
259  if (isnegative) {
260  *b++ = '-';
261  }
262 
263  char* e = b - 1;
264 
265  while(e > buffer) { // Do the reversal in-place instead of making temp buffer. Better cache locality, and we don't have to guess how big to make the temp buffer.
266  char tmp = *e;
267  *e = *buffer;
268  *buffer = tmp;
269  ++buffer;
270  --e;
271  }
272 
273  return b;
274 }

461  void push_back_trailer(bool calculate_checksum = true) {
462  // Calculate and write out the BodyLength.
463  // BodyLength does not include the SOH character after the BodyLength field.
464  // BodyLength does not include the SOH character before the CheckSum field.
465  size_t bodylength = next_ - (body_length_ + 7);
466  for(char* b = body_length_ + 5; b >= body_length_; --b) {
467  *b = '0' + (bodylength % 10);
468  bodylength /= 10;
469  }
470 
471  // write out the CheckSum after optionally calculating it
472  if (calculate_checksum) {
473  size_t checksum(0);
474  char* b = buffer_;
475  while(b < next_) checksum += *b++;
476  checksum = checksum % 256;
477 
478  memcpy(next_, "10=", 3);
479  next_ += 3;
480 
481  for (char* b = next_ + 2; b >= next_; --b) {
482  *b = '0' + (checksum % 10);
483  checksum /= 10;
484  }
485  next_ += 3;
486  *next_++ = '\x01';
487  } else {
488  memcpy(next_, "10=000\x01", 7);
489  next_ += 7;
490  }

1862
