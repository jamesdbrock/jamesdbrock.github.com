126 \brief Internal integer-to-ascii conversion.
127 
128 Writes an integer out as ascii.
129 
130 \tparam Int_type Type of integer to be converted.
131 \param number Value of the integer to be converted.
132 \param buffer Pointer to location for the ascii to be written.
133 \param end Past-the-end of the buffer, to check for overflow.
134 \return Pointer to past-the-end of the ascii that was written.
135 */
136 template<typename Int_type> char* itoa(Int_type number, char* buffer, char* end)
137 {
138  // Write out the digits in reverse order.
139  bool isnegative(false);
140  if (number < 0) {
141  isnegative = true;
142  number = -number;
143  }
144 
145  char*b = buffer;
146  do {
147  if (b >= end) details::throw_range_error();
148  *b++ = '0' + (number % 10);
149  number /= 10;
150  } while(number);
151 
152  if (isnegative) {
153  if (b >= end) details::throw_range_error();
154  *b++ = '-';
155  }
156 
157  // Reverse the digits in-place.
158  std::reverse(buffer, b);
159 
160  return b;
161 }

192 \brief Internal ascii-to-decimal conversion.
193 
194 Converts an ascii string to a decimal float, of the form \htmlonly mantissa&times;10<sup>exponent</sup>\endhtmlonly.
195 
196 Non-normalized. The exponent will always be less than or equal to zero.
197 If the decimal float is an integer, the exponent will be zero.
198 
199 \tparam Int_type Signed integer type for the mantissa and exponent.
200 \param begin Pointer to the beginning of the ascii string.
201 \param end Pointer to past-the-end of the ascii string.
202 \param mantissa Reference to storage for the mantissa of the decimal float to be returned.
203 \param exponent Reference to storage for the exponent of the decimal float to be returned.
204 */
205 template<typename Int_type> void atod(char const* begin, char const* end, Int_type& mantissa, Int_type& exponent)
206 {
207  Int_type mantissa_ = 0;
208  Int_type exponent_ = 0;
209  bool isdecimal(false);
210  bool isnegative(false);
211 
212  if (begin < end && *begin == '-') {
213  isnegative = true;
214  ++begin;
215  }
216 
217  for(; begin < end; ++begin) {
218  if (*begin == '.') {
219  isdecimal = true;
220  } else {
221  mantissa_ *= 10;
222  mantissa_ += (*begin - '0');
223  if (isdecimal) --exponent_;
224  }
225  }
226 
227  if (isnegative) mantissa_ = -mantissa_;
228  mantissa = mantissa_;
229  exponent = exponent_;
230 }

232 /*
233 \brief Internal decimal-to-ascii conversion.
234 
235 Converts a decimal float of the form \htmlonly mantissa&times;10<sup>exponent</sup>\endhtmlonly to ascii.
236 
237 Non-normalized. The exponent parameter must be less than or equal to zero.
238 
239 \tparam Int_type Integer type for the mantissa and exponent.
240 \param mantissa The mantissa of the decimal float.
241 \param exponent The exponent of the decimal float. Must be less than or equal to zero.
242 \param buffer Pointer to location for the ascii to be written.
243 \param end Past-the-end of the buffer, to check for overflow.
244 \return Pointer to past-the-end of the ascii that was written.
245 */
246 template<typename Int_type> char* dtoa(Int_type mantissa, Int_type exponent, char* buffer, char* end)
247 {
248  // Write out the digits in reverse order.
249  bool isnegative(false);
250  if (mantissa < 0) {
251  isnegative = true;
252  mantissa = -mantissa;
253  }
254 
255  char*b = buffer;
256  do {
257  if (b >= end) details::throw_range_error();
258  *b++ = '0' + (mantissa % 10);
259  mantissa /= 10;
260  if (++exponent == 0) {
261  if (b >= end) details::throw_range_error();
262  *b++ = '.';
263  }
264  } while(mantissa > 0 || exponent < 1);
265 
266  if (isnegative) {
267  if (b >= end) details::throw_range_error();
268  *b++ = '-';
269  }
270 
271  // Reverse the digits in-place.
272  std::reverse(buffer, b);
273 
274  return b;
275 }
