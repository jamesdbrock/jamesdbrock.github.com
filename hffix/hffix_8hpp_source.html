130 \brief Internal integer-to-ascii conversion.
131 
132 Writes an integer out as ascii.
133 
134 \tparam Int_type Type of integer to be converted.
135 \param number Value of the integer to be converted.
136 \param buffer Pointer to location for the ascii to be written.
137 \param end Past-the-end of the buffer, to check for overflow.
138 \return Pointer to past-the-end of the ascii that was written.
139 */
140 template<typename Int_type> char* itoa(Int_type number, char* buffer, char* end)
141 {
142  // Write out the digits in reverse order.
143  bool isnegative(false);
144  if (number < 0) {
145  isnegative = true;
146  number = -number;
147  }
148 
149  char*b = buffer;
150  do {
151  if (b >= end) details::throw_range_error();
152  *b++ = '0' + (number % 10);
153  number /= 10;
154  } while(number);
155 
156  if (isnegative) {
157  if (b >= end) details::throw_range_error();
158  *b++ = '-';
159  }
160 
161  // Reverse the digits in-place.
162  std::reverse(buffer, b);
163 
164  return b;
165 }

196 \brief Internal ascii-to-decimal conversion.
197 
198 Converts an ascii string to a decimal float, of the form \htmlonly mantissa&times;10<sup>exponent</sup>\endhtmlonly.
199 
200 Non-normalized. The exponent will always be less than or equal to zero.
201 If the decimal float is an integer, the exponent will be zero.
202 
203 \tparam Int_type Signed integer type for the mantissa and exponent.
204 \param begin Pointer to the beginning of the ascii string.
205 \param end Pointer to past-the-end of the ascii string.
206 \param mantissa Reference to storage for the mantissa of the decimal float to be returned.
207 \param exponent Reference to storage for the exponent of the decimal float to be returned.
208 */
209 template<typename Int_type> void atod(char const* begin, char const* end, Int_type& mantissa, Int_type& exponent)
210 {
211  Int_type mantissa_ = 0;
212  Int_type exponent_ = 0;
213  bool isdecimal(false);
214  bool isnegative(false);
215 
216  if (begin < end && *begin == '-') {
217  isnegative = true;
218  ++begin;
219  }
220 
221  for(; begin < end; ++begin) {
222  if (*begin == '.') {
223  isdecimal = true;
224  } else {
225  mantissa_ *= 10;
226  mantissa_ += (*begin - '0');
227  if (isdecimal) --exponent_;
228  }
229  }
230 
231  if (isnegative) mantissa_ = -mantissa_;
232  mantissa = mantissa_;
233  exponent = exponent_;
234 }

236 /*
237 \brief Internal decimal-to-ascii conversion.
238 
239 Converts a decimal float of the form \htmlonly mantissa&times;10<sup>exponent</sup>\endhtmlonly to ascii.
240 
241 Non-normalized. The exponent parameter must be less than or equal to zero.
242 
243 \tparam Int_type Integer type for the mantissa and exponent.
244 \param mantissa The mantissa of the decimal float.
245 \param exponent The exponent of the decimal float. Must be less than or equal to zero.
246 \param buffer Pointer to location for the ascii to be written.
247 \param end Past-the-end of the buffer, to check for overflow.
248 \return Pointer to past-the-end of the ascii that was written.
249 */
250 template<typename Int_type> char* dtoa(Int_type mantissa, Int_type exponent, char* buffer, char* end)
251 {
252  // Write out the digits in reverse order.
253  bool isnegative(false);
254  if (mantissa < 0) {
255  isnegative = true;
256  mantissa = -mantissa;
257  }
258 
259  char*b = buffer;
260  do {
261  if (b >= end) details::throw_range_error();
262  *b++ = '0' + (mantissa % 10);
263  mantissa /= 10;
264  if (++exponent == 0) {
265  if (b >= end) details::throw_range_error();
266  *b++ = '.';
267  }
268  } while(mantissa > 0 || exponent < 1);
269 
270  if (isnegative) {
271  if (b >= end) details::throw_range_error();
272  *b++ = '-';
273  }
274 
275  // Reverse the digits in-place.
276  std::reverse(buffer, b);
277 
278  return b;
279 }

528  // Calculate and write out the BodyLength.
